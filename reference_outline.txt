192	Normal	align=None	def user_col_filter(user_name,user_ratings,top_n=5):
    # 获得传入的用户收藏的电影
    target_user_ratings = user_ratings[user_name]
    if not target_user_ratings:
        return []
    else:
        user_similarity_scopes = {}
        # 转变为np
        target_user_ratings_list = np.array([
            ratings for _,ratings in target_user_ratings.items()
        ])
        for user,ratings in user_ratings.items():
            # 获取除了该用户之外的其余用户收藏的电影，当用户与所传入的用户一致时，跳过该循环
            if user == user_name:
                continue
            # 将其余用户的收藏电影转化为np
            user_ratings_list = np.array([
                ratings.get(item,0) for item in target_user_ratings
            ])
            # 计算传入的用户与其余用户的相关分数，并存入user_similarity_scopes字典中
            similarity_score = cosine_similarity([user_ratings_list,target_user_ratings_list])[0][0]
            user_similarity_scopes[user] = similarity_score
        # 对相关分数进行排序，获取
        sorted_similarity_user = sorted(user_similarity_scopes.items(),key=lambda x:x[1],reverse=True)
        recommend_items = set()
        # 只取其中与所传入用户相似度最高的top_n个用户所收藏的电影，并通过集合去除掉其中重复的
        for similar_user,_ in sorted_similarity_user[:top_n]:
            recommend_items.update(user_ratings[similar_user].keys())
        # 排除用户自身已经收藏的电影，剩下的就是推荐的电影
        recommend_items = [item for item in recommend_items if item not in target_user_ratings]
        print(recommend_items)
        return recommend_items
